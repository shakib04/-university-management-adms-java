
//Course
CREATE TABLE  "COURSE" 
   (	"COURSE_ID" VARCHAR2(4000), 
	"COURSE_NAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"SEMESTER" VARCHAR2(4000) NOT NULL ENABLE, 
	"DEPT_NAME" VARCHAR2(4000) NOT NULL ENABLE, 
	 CONSTRAINT "COURSE_PK" PRIMARY KEY ("COURSE_ID") ENABLE
   )

//Employee
CREATE TABLE  "EMPLOYEE" 
   (	"EMPID" VARCHAR2(4000) NOT NULL ENABLE, 
	"EMPLOYEENAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"DESIGNATION" VARCHAR2(4000) NOT NULL ENABLE, 
	"SALARY" NUMBER(20,2) NOT NULL ENABLE, 
	"STATUS" VARCHAR2(4000) NOT NULL ENABLE, 
	 CONSTRAINT "EMPLOYEE_PK" PRIMARY KEY ("EMPID") ENABLE
   )

//Person
CREATE TABLE  "PERSON" 
   (	"PERSONID" VARCHAR2(4000) NOT NULL ENABLE, 
	"PASSWORD" VARCHAR2(4000) NOT NULL ENABLE, 
	"STATUS" VARCHAR2(4000) NOT NULL ENABLE, 
	 CONSTRAINT "PERSON_PK" PRIMARY KEY ("PERSONID") ENABLE
   )

//Section

CREATE TABLE  "SECTION" 
   (	"SECTION_ID" VARCHAR2(4000) NOT NULL ENABLE, 
	"COURSE_NAME" VARCHAR2(4000) NOT NULL ENABLE, 
	 CONSTRAINT "SECTION_PK" PRIMARY KEY ("SECTION_ID") DISABLE
   )

//Student
CREATE TABLE  "STUDENT" 
   (	"STU_ID" VARCHAR2(4000) NOT NULL ENABLE, 
	"STUDENTNAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"STU_DEPTNAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"CGPA" NUMBER(10,2) NOT NULL ENABLE, 
	"CREDIT_COMPLETE" NUMBER(10,2) NOT NULL ENABLE, 
	"CONTACT_NO" VARCHAR2(4000) NOT NULL ENABLE, 
	"STATUS" VARCHAR2(4000) NOT NULL ENABLE, 
	"SEMESTER_NAME" VARCHAR2(4000) NOT NULL ENABLE, 
	 CONSTRAINT "STUDENT_PK" PRIMARY KEY ("STU_ID") ENABLE
   )

//Teacher
CREATE TABLE  "TEACHER" 
   (	"TEACHERID" VARCHAR2(4000) NOT NULL ENABLE, 
	"TEACHERNAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"TEACHER_DEPTNAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"SALARY" NUMBER(30,2) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(4000) NOT NULL ENABLE, 
	"STATUS" VARCHAR2(4000) NOT NULL ENABLE, 
	 CONSTRAINT "TEACHER_PK" PRIMARY KEY ("TEACHERID") ENABLE
   )
